# HOW TO START!

    $  virtualenv -p python3 env
    $  source env/bin/activate
    $  pip install -r requirement.txt
    $  export DJANGO_SETTINGS_MODULE=config.settings
    $  python manage.py migrate
    $  python manage.py runserver 0.0.0.0:8000
